-- 9. CONTARE QUANTI ISCRITTI CI SONO STATI OGNI ANNO
SELECT COUNT(*) FROM STUDENTS;
SELECT
	YEAR(ENROLMENT_DATE) AS YEAR_ENROLMENT,
	COUNT(ID) AS ENROLMENT
FROM
	STUDENTS
GROUP BY
	YEAR(ENROLMENT_DATE);
	

-- 10. CALCOLARE GLI INSEGNATI CHE HANNO L'UFFICIO NELLO STESSO EDIFICIO
SELECT
	OFFICE_ADDRESS ,
	COUNT(*) AS TEACHERS
FROM
	TEACHERS
GROUP BY
	OFFICE_ADDRESS;
	

-- 11. CALCOLARE LA MEDIA DEI VOTI DI OGNI APPELLO D'ESAME
SELECT
	EXAM_ID,
	AVG(VOTE) AS AVARAGE_VOTE
FROM
	EXAM_STUDENT
GROUP BY
	EXAM_ID;
	

-- 12 CONTARE QUANTI CORSI DI LAUREA CI SONO PER OGNI DIPARTIMENTO
SELECT
	DEGREES.DEPARTMENT_ID AS DEPARTMENTS,
	COUNT(COURSES.ID) AS COURSES
FROM
	COURSES
INNER JOIN DEGREES ON
	COURSES.ID = DEGREES.ID
GROUP BY
	(DEGREES.DEPARTMENT_ID);
